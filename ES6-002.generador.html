Generadores - Rangos

function * rango (inicio, fin, incremento){

while(inicio < fin){

yield inicio;

inicio += incremento;

}

}

for(let i of rango(0, 10, 2)){

console.log(i);

}

Generadores - Cuadrados

function* cuadrados(){

let n = 1;

while(true){

let c = n*n;

n++;

yield c;

}

}
